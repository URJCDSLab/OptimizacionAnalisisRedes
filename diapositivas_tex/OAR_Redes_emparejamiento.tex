\begin{frame}{Problema de emparejamiento}
	\begin{block}{Definición}
		Dado un grafo no dirigido $G=(V,E)$, un \alert{emparejamiento} es un subconjunto $K \subset E$ que no contiene dos aristas adyacentes, es decir, con un extremo en com\'{u}n.
	\end{block}
	\begin{block}{Proposición}
		Dado un grafo $G=(V,E)$ y un emparejamiento $K$ de $G$, cualquier subconjunto de $K$ es también un emparejamiento de $G$.
	\end{block}
	\alert{Demostración:} trivial. 
	\begin{block}{Problema de emparejamiento de máximo cardinal}
		Dado $G$, determinar el emparejamiento $K$ de cardinal máximo. 
	\end{block}
\end{frame}

\section{Resolución}

\begin{frame}{Emparejamiento}
	\begin{block}{Definición}
		Se dice que un v\'{e}rtice $x \in V$ est\'{a} \alert{saturado} por el emparejamiento $K$ si existe una arista $\{ x,y \} \in K$.
		
		En caso contrario, se dice que el v\'{e}rtice est\'{a} \alert{expuesto} o no saturado.
	\end{block}
	\begin{block}{Proposición}
		Dado un grafo $G=(V,E)$ y un emparejamiento $K$ de $G$, si el número de vértices no saturados es 0 o 1, el emparejamiento es máximo.
	\end{block}
	\alert{Demostración:} trivial. 
	\begin{block}{Definición}
		Si $G$ no tiene vértices no saturados, se dice que el emparejamiento es perfecto.
	\end{block}
\end{frame}

\begin{frame}{Emparejamientos}
	
	\begin{block}{Definición}
		Dados un grafo $G=(V,E)$ y un emparejamiento $K \subset E$, se define una \alert{cadena} \alert{alternante} como una cadena en la que las aristas pertenecen y no pertenecen alternativamente al emparejamiento.
	\end{block}
	\begin{block}{Definición}
		Una cadena alternante $\mu[x,y]$, con $x \neq y$, se dice \alert{incremental} si sus extremos $x$ e $y$ son no saturados por el emparejamiento $K$.
	\end{block}
	\begin{itemize}
		\item Una cadena incremental tiene un n\'{u}mero impar de aristas. 
		\item El n\'{u}mero de aristas que no pertenecen al emparejamiento siempre es una unidad mayor que el n\'{u}mero de aristas del emparejamiento.
	\end{itemize}
\end{frame}

\begin{frame}{Cadena incremental y transferencia}
	\begin{itemize}
		\item  Las cadenas incrementales permiten aumentar el cardinal del emparejamiento en una unidad mediante una \alert{transferencia}.
		\item La transferencia en una cadena incremental $\mu[x,y]$ respecto del emparejamiento $K$ consiste en permutar la pertenencia y no pertenencia de las aristas de la cadena a dicho emparejamiento.
	\end{itemize}
	\begin{block}{Transferencia}
		Dado un emparejamiento $K$ y una cadena incremental $\mu[x,y]$ respecto del mismo, la transferencia es la diferencia sim\'{e}trica entre la cadena y el emparejamiento:
		$K' = K \bigtriangleup \mu[x,y]$. Se verifica que $\vert K' \vert = \vert K \vert + 1 $.
	\end{block}	
	\footnotesize
	\alert{Nota:} la diferencia sim\'{e}trica de dos conjuntos $A \bigtriangleup B$ es el conjunto de elementos que pertenecen exclusivamente a uno solo de los dos conjuntos: 
	$A \bigtriangleup B \equiv \{ x \in A \cup B \ / \ x \not\in A \cap B \}$
\end{frame}

\begin{frame}{Emparejamientos}
	\begin{block}{Teorema}
		Dado un grafo $G=(V,E)$, un emparejamiento $K \subset E$ es de m\'{a}ximo cardinal si y solo si no existe ninguna cadena incremental respecto de $K$.
	\end{block}
	\begin{block}{Lema}
		Dados dos emparejamientos $K_1$ y $K_2$ en un grafo simple $G=(V,E)$, al considerar el grafo parcial $G'=(V,K_1 \bigtriangleup K_2)$ se verifica que todas sus componentes conexas son de alguno de los tres tipos:
		\begin{itemize}
			\item V\'{e}rtice aislado
			\item Ciclo elemental alternante (con aristas de $K_1$ y $K_2$ alternativamente). Un ciclo alternante tiene necesariamente un n\'{u}mero par de aristas.
			
			\item Cadena elemental alternante. Una cadena alternante puede tener un n\'{u}mero par o impar de aristas.
		\end{itemize}
	\end{block}
	\alert{Nota:} alternante $(K_1,K_2)$ indica que las aristas pertenecen alternativamente a $K_1$ y $K_2$.
	
\end{frame}
\begin{frame}{Emparejamientos}
	\alert{Demostración (lema):}
	
	Analicemos qué tipo de vértices podemos tener en el grafo $G$:
	\begin{itemize}
		\item Nodos no saturados por ninguno de los emparejamientos: serán de grado 0 en $G'$. Son vértices aislados.
		\item Nodos saturados solo por uno de los emparejamientos: serán de grado 1 en $G'$ y, por tanto, extremos en las cadenas alternantes.
		\item Nodos saturados por los dos emparejamientos:
		\begin{itemize}
			\item Saturados por la misma arista en los dos emparejamientos: serán de grado 0 en $G'$ y, por tanto, son vértices aislados. 	
			\item Saturados por distintas aristas en cada caso: serán vértices de grado 2 en $G'$ y, por tanto, formarán parte de un ciclo alternante o del interior de una cadena alternante. 
		\end{itemize}
	\end{itemize}
	Por tanto, en $G'$ no hay vértices de grado mayor que 2. Las únicas estructuras posibles son las indicadas en el enunciado del lema.
\end{frame}

\begin{frame}{Emparejamientos}
	\begin{block}{Comprensión de un grafo}
		Dado un grafo $G=(V,E)$ y un subconjunto de vértices $W\subset V$, se define el grafo comprimnido $G/W$ como:
		$$
		G/W =\bigl( V\setminus W\cup\{w\}, E/W\bigr)
		$$
		donde $E/W$ es el conjunto de aristas
		$$
		E/W=\bigl\{\{i,j\}\,\vert\, i,j\notin W\bigr\}\cup \bigl\{\{w,j\}\,\vert\, \{j,i\}\in E,  j\in W, i\notin W\bigl\}
		$$
	\end{block}
\end{frame}
\begin{frame}{Emparejamientos}
	\begin{block}{Teorema}
		Sea $G=(V,E)$ un grafo, $W\subset V$ un subconjunto de vértices asociado a un ciclo impar y $K_1$ un emparejamiento en el grafo comprimido $G/W$. Entonces, existe un emparejamiento $K_W$ en el subgrafo generado por $W$  tal que $K=K_1\cup K_W$ es un emparejamiento de $G$.
	\end{block}
	Nota: 
	\begin{itemize}
		\item La clave está en que al ser un ciclo impar, siempre queda un vértice no saturado en $W$ respecto de $K_W$.
		\item Siempre se puede elegir qué vértice del ciclo es el que se queda no saturado
		\item Si el vértice asociado al ciclo comprimido está saturado en $K_1$, se deja como no saturado en el ciclo precisamente el vértice asociado a la arista que satura al vértice $w$.
	\end{itemize}
\end{frame}

\subsection{\'{A}rbol alternante}

\begin{frame}{Árbol alternante}
	
	\begin{itemize}\itemsep 3mm
		\item Dado un emparejamiento y un v\'{e}rtice expuesto, se trata de encontrar una cadena incremental impar que parta de dicho v\'{e}rtice.
		\item Los v\'{e}rtices $x \in V$ saturados por el emparejamiento se identificar\'{a}n por $sat(x)=1$; los v\'{e}rtices expuestos o no saturados verificar\'{a}n $sat(x)=0$.
		\item A partir de un emparejamiento,  se construye un conjunto de \'{a}rboles alternantes inconexos, que se puede definir como un bosque alternante.
		\item \alert{Nota:} a partir de un emparejamiento en un grafo, bipartito o no, siempre se puede construir un grafo bipartito a partir del grafo parcial que resulta de seleccionar las aristas
		del emparejamiento.
	\end{itemize}
\end{frame}




\begin{frame}{Árbol alternante}
	
	\begin{block}{Definición}
		Dados un grafo $G=(V,E)$ y un emparejamiento $K \subset E$, un 	\alert{\'{a}rbol alternante} es un subgrafo parcial $H=(W,T)$, con 	$W \subset V$ y $T \subset E$ con las siguientes propiedades:
		\begin{enumerate}
			\item $H$ es un \'{a}rbol.
			\item Existe un v\'{e}rtice $r \in W$, que se denomina ra\'{\i}z del \'{a}rbol alternante,
			y que es un v\'{e}rtice expuesto respecto del emparejamiento $K$.
			\item Cualquier cadena elemental que une un v\'{e}rtice $w \in W$  con la ra\'{\i}z $r$ es una
			cadena alternante.
		\end{enumerate}
	\end{block}

\begin{frame}{Árbol alternante}
	
	\begin{itemize}\itemsep 5mm
		\item Sea $R \subset V$ el conjunto de las ra\'{\i}ces del bosque alternante que se construye en el grafo $G=(V,E)$ a partir de un 	emparejamiento $K \subset E$. 
		\item El bosque alternante se caracteriza por el vector ${\bf p}$ (indica, para cada vértice, cuál es su padre).
		\item Cada ra\'{\i}z $r \in R$ de un \'{a}rbol alternante $H(r)=(W(r),T(r))$ se identifica por $p(r)=r$. 
		\item La \'{u}nica cadena alternante que une cualquier v\'{e}rtice $w \in W(r)$ del \'{a}rbol con su ra\'{\i}z $r$ es: 
		$$ 
		w \longleftrightarrow  p(w)  \longleftrightarrow  p(p(w)) \longleftrightarrow \cdots  \longleftrightarrow p(\cdots(p(w)) \cdots)  \longleftrightarrow  r 
		$$
	\end{itemize}
	
\end{frame}

\begin{frame}{Emparejamientos. Algoritmo}
	\begin{description}\itemsep 3mm
		\item[Paso A.-] Encontrar un emparejamiento $K$ inicial (por ejemplo $K=\emptyset$).
		\item[Paso B.-] Si el número de vértices no saturados es 0 o 1, PARAR. En otro caso ir al siguiente paso.
		\item[Paso C.-] Construir un árbol alternante asociado a $K$:
		\begin{enumerate}\itemsep 2mm
			\item Inicializar el árbol alternante.
			\item Construir el árbol alternante:
			\begin{description}\itemsep 1mm
				\item[Caso 1:] Se detecta un ciclo impar, se contrae (modificando el grafo) y se va al  al paso C.2.
				\item[Caso 2:] Se amplía el árbol alternante y se va al paso C.2.
				\item[Caso 3:] Se detecta una cadena alternante. Se incrementa $K$ y se va al paso B.
				\item[Caso 4:] En el resto de los casos, PARAR, $K$ es maximal.
			\end{description}
		\end{enumerate}
	\end{description}
\end{frame}

\subsection{Recubrimiento en un grafo}

\begin{frame}{Recubrimiento}
	El problema de recubrimiento está relacionado con el problema de emparejamiento
	\begin{itemize}
		\item  Emparejamiento: subconjunto de aristas con la propiedad de no haber aristas adyacentes.
		\item Recubrimiento: subconjunto de aristas con la propiedad de ser todos los v\'{e}rtices incidentes en, al menos, alguna de las aristas seleccionadas.
		\item Es decir, es un problema similar al de emparejamiento, pero ahora todos los vértices tienen que estar saturados.
	\end{itemize}
	\begin{block}{Recubrimiento}
		Dado un grafo $G=(V,E)$, un \alert{recubrimiento} es un subconjunto de aristas $F \subset E$ de forma que cualquier v\'{e}rtice del grafo es incidente en alguna arista de $F$.
	\end{block}
	Nota: en grafos conexos el propio conjunto $E$ es un recubrimiento
\end{frame}

\begin{frame}{Recubrimiento}
	
	\begin{block}{Proposición}
		Dado un grafo $G=(V,E)$ y  $F$ un recubrimiento de $G$, si $F'$ es tal que $F\subset F'$, $F'$ es un recubrimiento de $G$.
	\end{block}
	
	żCuál es el recubrimiento de mínimo cardinal (recubrimiento minimal o recubrimiento de mínimo cardinal)?
	
	Este problema está relacionado con el problema de emparejamiento de máximo cardinal.
\end{frame}


\begin{frame}{Cubrimiento}
	
	\begin{block}{Teorema}
		Sea un grafo simple $G=(V,E)$ con $\vert V\vert=n$, $K_0$ un emparejamiento de máximo cardinal y $F_0$ un recubrimiento de mínimo cardinal. Entonces:
		\begin{enumerate}
			\item $\vert K_0\vert + \vert F_0\vert =n$
			\item $F_1=K_0\cup \{$una arista incidente en cada vértice no saturado por $K_0\}$ es un recubrimiento de mínimo cardinal.
			\item $K_1=F_0\setminus \{$todas las aristas menos una de las que inciden en cada vértice saturado 2 o más veces por $F_0\}$ es un emparejamiento de máximo cardinal.
		\end{enumerate}
	\end{block}
	\begin{itemize}
		\item 	Este teorema permite resolver el problema de recubrimiento de mínimo cardinal a partir de la solución al problema de emparejamiento de máximo cardinal.
		\item  Solo se ańade un procedimiento de complejidad $\theta(n)$ al algoritmo de emparejamiento de máximo cardinal.
	\end{itemize}
\end{frame}

		\item Sea $K_0$ un emparejamiento de máximo cardinal. Entonces $K_0$ satura $2\vert K_0\vert$ vértices y quedan $n-2\vert K_0\vert$ vértices sin saturar.

\begin{frame}{Recubrimiento}
	\begin{itemize}\itemsep 3mm
		\item Sea $F_0$ un recubrimiento de mínimo cardinal. Todos los vértices están saturados y las cadenas de $F_0$ tienen longitud menor o igual a 2.
		\item Sea $K_1$ a partir de $F_0$: $\forall n: grado(n)>1$ en $F_0$, se quitan $grad(n)-1$ aristas.
		\item Es obvio que $K_1$ es un emparejamiento: ningún vértice tendrá grado mayor que 1. 
		\item En $F_0$ todos los vértices están saturados y en $K_1$ puede haber vértices no saturados. Cada arista que se elimina de $F_0$ deja de saturar exactamente un vértice.
		\item Por tanto, el número de vértices no saturados en $K_1$ es igual al número de aristas eliminadas de $F_0$:
		$$
		\vert K_1\vert = \vert F_0\vert -(n-2\vert K_1\vert)\Rightarrow  \vert F_0\vert + \vert K_1\vert =n
		$$
		
	\end{itemize}
	
\end{frame}

\begin{frame}{Recubrimiento}
	\begin{itemize}\itemsep 5mm
		\item 	żEs $K_1$ un emparejamiento de máximo cardinal y $F_1$ un recubrimiento de mínimo cardinal?
		\begin{itemize}\itemsep 2mm
			\item 		Por ser $K_1$ emparejamiento: $\vert K_1\vert \leq \vert K_0\vert$.
			\item Por ser $F_1$ recubrimiento: $\vert F_1\vert \geq \vert F_0\vert$.
		\end{itemize}
		\item Por tanto:
		$$
		n = \vert F_1\vert + \vert K_0\vert \geq  \vert F_1\vert + \vert K_1\vert \geq  \vert F_0\vert + \vert K_1 \vert = n
		$$
		\item Por tanto, $\vert F_1\vert = \vert F_0\vert $ y $\vert K_1\vert = \vert K_0\vert$ .
		\item Es decir, $K_1$ es un emparejamiento de máximo cardinal y $F_1$ es un recubrimiento minimo.
	\end{itemize}
\end{frame}

\section{Emparejamiento de máximo peso}

\begin{frame}{Emparejamiento de máximo peso}{Definición del problema}
	
	\begin{block}{Emparejamiento de máximo peso}
		Dado una grafo $G=(V,E)$ y un vector $p$ de pesos asociados a las aristas $\{p(e)\,\vert\, e\in E\})$ el problema de emparejamiento de máximo peso trata de determinar un subconjunto $K\subset E$ de aristas tal que:
		\begin{itemize}
			\item $K$ es un emparejamiento de $G$.
			\item Para cualquier otro emparejamiento $K'$, $p(K')\leq p(K)$.
		\end{itemize}
	\end{block}Este problema es una generalización del problema del emparejamiento de máximo cardinal. 
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}
	\begin{itemize}\itemsep 5mm
		\item En el problema de emparejamiento de máximo cardinal, la clave es la \alert{cadena alternante incremental}.
		\item En este problema, el objetivo no es incrementar el cardinal, sino el peso.
		\item En ocasiones, esto se puede lograr reduciendo el cardinal del emparejamiento.
		\item Por tanto, el concepto que hay que manejar es el de cadena alternante.
	\end{itemize}
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}{Definición del problema}
	\begin{block}{Definición}
		Dado un grafo $G=(V,E)$, unos pesos $p$ sobre las aristas, un emparejamiento $K$ de $G$ y una cadena alternante $L$, se define el coste reducido de $L$ respecto de $K$, $\delta_L$ como:
		$$
		\delta_L=p(L\cap\overline{K})-p(L\cap K)
		$$
		donde $\overline{K} =E\setminus K$. 
	\end{block}	
	\begin{itemize}
		\item Si $\delta_L >0$, la transferencia mejora.
		\item Si $\delta_L =0$, la transferencia no cambia el peso de la solución.
		\item Si $\delta_L <0$, la transferencia empeora  el peso de la solución.
	\end{itemize}
	\begin{block}{Definición}
		Un ciclo alternante es una cadena alternante par y cerrada.
	\end{block}	
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}
	\begin{block}{Teorema}
		Dado un grafo $G=(V,E)$ y unos pesos $p$ sobre las aristas, un emparejamiento $K$ de $G$ es de máximo peso si y solo si no existen cadenas alternantes con coste reducido estrictamente positivo.
	\end{block}
	\visible<2>{
	\alert{Demostración:}\small
	\begin{itemize}
		\item $\Rightarrow$: Trivial, ya que si existiera esta cadena, sería posible obtener un emparejamiento de mayor peso y $K$ no sería de máximo peso.
		\item $\Leftarrow$:  Sea $K_1$ un emparejamiento de máximo peso. Por tanto, $p(K_1)\geq p(K)$.  Veamos que $p(K_1)\leq p(K)$.
		
		Consideremos el grafo parcial $H=(V,E_H)$, donde $E_H=(K\setminus K_1)\cup (K_1\setminus K)$
		
		El grafo $H$ tiene componentes conexas de tres tipos: vértices aislados, cadenas alternantes y ciclos alternantes. 
		
		Los vértices aislados no influyen en la solución. Los ciclos alternantes y las cadenas alternantes respecto de $K$ no pueden tener coste reducido positivo. Por tanto, trivialmente $p(K_1)\leq p(K)$ y $K$ es una solución del problema. 
		
	\end{itemize}}
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}
	
	El algoritmo de resolución se basa en ir obteniendo los emparejamientos $K^m$ de máximo peso con exactamente $m$ aristas:
	\begin{itemize}\itemsep 2mm
		\item $K^1$ es la arista de máximo peso en $E$.
		\item $K^2$ es el par de aristas disjuntas (sin vértices en común) de máximo peso.
	\end{itemize}
	\begin{block}{Teorema}
		Sea $G=(V,E)$ un grafo, $K^m$ el emparejamiento de máximo peso de cardinal $m$ y sea $L$ una cadena alternante \alert{incremental} respecto de $K^m$ con coste reducido máximo. Entonces, el emparejamiento $K^{m+1}$ obtenido a partir de $K^m$ por la transferencia de $ L$ es el emparejamiento de máximo peso de cardinal $m+1$.
	\end{block}
	Nota: $K^{m+1}$ no tiene por qué ser el emparejamiento óptimo de $m+1$ o menos aristas. 
	
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}
	\alert{Demostración:}
	
	Sea $K^*$ el emparejamiento de máximo peso con $m+1$ aristas. Y sea $H=(V,E_H)$, donde $E_H=(K^*\setminus K^m)\cup (K^m\setminus K^*)$
	
	\bigskip
	
	El grafo $H$ tiene componentes conexas de tres tipos: vértices aislados, cadenas alternantes y ciclos alternantes. 
	\begin{itemize}\itemsep 5mm
		\item Los vértices aislados no influyen.
		\item Los ciclos alternantes deben ser tales que su coste reducido sea 0, ya que tanto $K^*$ como $K^m$ son óptimos. 
		\item Las cadenas alternantes pares también tiene coste reducido 0.
	\end{itemize}
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}
	\alert{Demostración (cont.):}
	
	Cadenas alternantes impares: decimos que domina el emparejamiento que tiene más aristas en la cadena.
	
	Como $K^*$ tiene una arista más que $K^m$, $K^*$ dominará una cadena más que $K^m$: si hay $s$ cadenas dominadas por $K^m$, habrá $s+1$ dominadas por $K^*$.
	
	Las $s$ cadenas dominadas por $K^m$ se pueden compensar con $s$ cadenas dominadas por $K^*$. 
	
	Además, todas estas cadenas {\em compensadas} tienen coste reducido 0.
	
	La única cadena {\em no compensada} , $L^*$, es una cadena alternante \alert{incremental}.
	
	Esta cadena concentra toda la diferencia en peso entre $K^*$ y $K^m$:
	$$
	w(K^*)=w(K^m)+\delta(L^*)
	$$
\end{frame}

\begin{frame}{Emparejamiento de máximo peso}
	\alert{Demostración (cont.):}
	
	Esta cadena concentra toda la diferencia en peso entre $K^*$ y $K^m$:
	$$
	w(K^*)=w(K^m)+\delta(L^*)
	$$
	Por otro lado, 
	$$
	w(K^{m+1})=w(K^m)+\delta(L)
	$$
	Como $K^*$ es el emparejamiento de peso máximo con $m+1$ aristas:
	$$
	w(K^{m+1})\leq w(K^*)\quad\Rightarrow\quad  \delta(L^*) \geq \delta(L)
	$$
	Por tanto, 
	$$
	\delta(L^*) = \delta(L)
	$$
	y $K^{m+1}$ es un emparejamiento de peso máximo con $m+1$ aristas.
\end{frame}
